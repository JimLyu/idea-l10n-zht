報告 NewType 的無效用法。

範例： 


  from typing import NewType

  InvalidName = NewType("Name", int)  # 變數名稱 'InvalidName' 與 NewType 名稱 'Name' 不符合


  from typing import Literal

  InvalidType = NewType("InvalidType", Literal[1])  # NewType 不能與 'Literal[1]' 一起使用


  Base = NewType("Base", str)

  class Derived(Base):  # 'Base' cannot be subclassed
      pass
